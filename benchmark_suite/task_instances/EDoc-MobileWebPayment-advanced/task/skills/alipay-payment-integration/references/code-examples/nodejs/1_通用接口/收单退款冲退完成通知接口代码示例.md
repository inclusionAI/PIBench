# 收单退款冲退完成通知接口 - NODEJS 示例

## 请求示例

```javascript
const express = require('express');
const { AlipaySdk } = require("alipay-sdk");

const app = express();
app.use(express.urlencoded({ extended: true }));

const alipaySdk = new AlipaySdk({
  appId: "<-- 请填写您的AppId，例如：2019091767145019 -->",
  privateKey: "<-- 请填写您的应用私钥，例如：MIIEvQIBADANB ... ... -->",
  alipayPublicKey: "<-- 请填写您的支付宝公钥，例如：MIIBIjANBg... -->",
});

// 商户服务端接收支付宝异步通知（alipay.trade.refund.depositback.completed）的处理示例
app.post('/notify', async (req, res) => {
  const signVerified = alipaySdk.checkNotifySign(req.body);

  if (signVerified) {
    // 验签通过，解析业务参数
    const data = JSON.parse(req.body.biz_content);

    // 处理业务逻辑...

    // 必须返回纯文本 success 给支付宝，否则支付宝会重发通知
    res.send('success');
  } else {
    // 验签失败
    res.send('fail');
  }
});

app.listen(3000, () => {
  console.log('服务已启动，监听端口 3000');
});

```
