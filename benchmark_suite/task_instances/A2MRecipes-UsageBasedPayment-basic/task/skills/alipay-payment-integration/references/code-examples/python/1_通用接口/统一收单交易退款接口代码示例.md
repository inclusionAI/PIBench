# 统一收单交易退款接口 - PYTHON 示例

## 请求示例

```python
import json
from alipay.aop.api.AlipayClientConfig import AlipayClientConfig
from alipay.aop.api.DefaultAlipayClient import DefaultAlipayClient
from alipay.aop.api.request.AlipayTradeRefundRequest import AlipayTradeRefundRequest
from alipay.aop.api.domain.AlipayTradeRefundModel import AlipayTradeRefundModel
from alipay.aop.api.domain.RefundGoodsDetail import RefundGoodsDetail
from alipay.aop.api.domain.OpenApiRoyaltyDetailInfoPojo import OpenApiRoyaltyDetailInfoPojo

# 初始化SDK
config = AlipayClientConfig()
config.server_url = "https://openapi.alipay.com/gateway.do"
config.app_id = "<-- 请填写您的AppId，例如：2019091767145019 -->"
config.app_private_key = "<-- 请填写您的应用私钥，例如：MIIEvQIBADANB ... ... -->"
config.alipay_public_key = "<-- 请填写您的支付宝公钥，例如：MIIBIjANBg... -->"
config.charset = "utf-8"
config.sign_type = "RSA2"
client = DefaultAlipayClient(alipay_client_config=config)

# 构造请求参数以调用接口
request = AlipayTradeRefundRequest()
model = AlipayTradeRefundModel()

# 设置商户订单号
model.out_trade_no = "20150320010101001"

# 设置支付宝交易号
model.trade_no = "2014112611001004680073956707"

# 设置退款金额
model.refund_amount = "200.12"

# 设置退款原因说明
model.refund_reason = "正常退款"

# 设置退款请求号
model.out_request_no = "HZ01RF001"

# 设置退款包含的商品列表信息
goods_detail = RefundGoodsDetail()
goods_detail.out_sku_id = "outSku_01"
goods_detail.out_item_id = "outItem_01"
goods_detail.goods_id = "apple-01"
goods_detail.refund_amount = "19.50"
goods_detail.out_certificate_no_list = ["202407013232143241231243243423"]
model.refund_goods_detail = [goods_detail]

# 设置退分账明细信息
royalty_info = OpenApiRoyaltyDetailInfoPojo()
royalty_info.amount = "0.1"
royalty_info.trans_in = "2088101126708402"
royalty_info.royalty_type = "transfer"
royalty_info.trans_out = "2088101126765726"
royalty_info.trans_out_type = "userId"
royalty_info.royalty_scene = "达人佣金"
royalty_info.trans_in_type = "userId"
royalty_info.trans_in_name = "张三"
royalty_info.desc = "分账给2088101126708402"
model.refund_royalty_parameters = [royalty_info]

# 设置查询选项
model.query_options = ["refund_detail_item_list"]

# 设置针对账期交易
model.related_settle_confirm_no = "2024041122001495000530302869"

request.biz_model = model
# 第三方代调用模式下请设置app_auth_token
# request.add_other_text_param("app_auth_token", "<-- 请填写应用授权令牌 -->")

response_content = client.execute(request)
print(response_content)

# 解析响应结果
response = json.loads(response_content)
result = response.get("alipay_trade_refund_response", {})
if result.get("code") == "10000":
    print("调用成功")
else:
    print("调用失败")
```
