# 收单退款冲退完成通知接口 - PYTHON 示例

## 请求示例

```python
import json
from flask import Flask, request
from alipay.aop.api.util.SignatureUtils import verify_with_rsa

app = Flask(__name__)

# 支付宝公钥
alipay_public_key = "<-- 请填写您的支付宝公钥，例如：MIIBIjANBg... -->"


# 商户服务端接收支付宝异步通知（alipay.trade.refund.depositback.completed）的处理示例
@app.route("/notify", methods=["POST"])
def notify():
    params = request.form.to_dict()

    # 从参数中提取签名信息
    sign = params.pop("sign", None)
    params.pop("sign_type", None)

    # 构造待验签字符串
    sorted_params = sorted([(k, v) for k, v in params.items() if v])
    message = "&".join(f"{k}={v}" for k, v in sorted_params)

    # 使用支付宝公钥验签
    sign_verified = verify_with_rsa(alipay_public_key, message, sign)

    if sign_verified:
        # 验签通过，解析业务参数
        biz_content = json.loads(params.get("biz_content", "{}"))

        # 处理业务逻辑...

        # 必须返回纯文本 success 给支付宝，否则支付宝会重发通知
        return "success"
    else:
        # 验签失败
        return "fail"


if __name__ == "__main__":
    print("服务已启动，监听端口 3000")
    app.run(host="0.0.0.0", port=3000)
```
